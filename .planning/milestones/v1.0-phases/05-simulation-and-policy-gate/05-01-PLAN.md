    ```rust
    use crate::EvmError;
    use alloy_dyn_abi::{DynSolType, DynSolValue};
    use alloy_json_abi::JsonAbi;
    use alloy_primitives::Bytes;
    use std::borrow::Cow;

    /// Encode a function call's input parameters into ABI calldata bytes.
    ///
    /// Performs the full Phase-4 dry-run encoding flow (parse JsonAbi →
    /// resolve overload → walk args via `js_value_to_dyn_sol` → call
    /// `Function::abi_encode_input`) and **returns the encoded bytes** (4-byte
    /// selector + tail-encoded args). Phase-4 `dry_run_abi_encode` discards
    /// these bytes; Phase-5 `normalize::normalize_contract_call` keeps them.
    ///
    /// Error categories propagate from the existing helpers — same wire-safe
    /// taxonomy as Phase 4 (Phase-4 BR-01 Cow<'static, str> compatible).
    pub fn encode_call_input(
        abi: &str,
        function: &str,
        args: &[serde_json::Value],
    ) -> Result<Bytes, EvmError> {
        // 1. Parse JsonAbi.
        let parsed: JsonAbi = serde_json::from_str(abi).map_err(|e| EvmError::Encode {
            category: Cow::Borrowed("abi_parse"),
            detail_for_log: format!("JsonAbi parse: {e}"),
        })?;

        // 2. Resolve overload by (name, arg count). Mirrors Phase-4 logic.
        let funcs = parsed.function(function).ok_or_else(|| EvmError::Encode {
            category: Cow::Borrowed("abi_function_not_found"),
            detail_for_log: format!("abi has no function `{function}`"),
        })?;
        let func = funcs.iter().find(|f| f.inputs.len() == args.len()).ok_or_else(|| EvmError::Encode {
            category: Cow::Borrowed("abi_overload_arity"),
            detail_for_log: format!(
                "no overload of `{function}` matches arg count {}",
                args.len()
            ),
        })?;

        // 3. Walk args via Phase-4 js_value_to_dyn_sol (unchanged).
        let dyn_values: Vec<DynSolValue> = func
            .inputs
            .iter()
            .zip(args.iter())
            .map(|(p, a)| {
                let ty: DynSolType = p.selector_type_name().parse().map_err(|e: alloy_dyn_abi::parser::Error| {
                    EvmError::Encode {
                        category: Cow::Borrowed("abi_type_parse"),
                        detail_for_log: format!("{}: {e}", p.selector_type_name()),
                    }
                })?;
                crate::dyn_abi::js_value_to_dyn_sol(a, &ty)
            })
            .collect::<Result<_, _>>()?;

        // 4. abi_encode_input prepends the 4-byte selector and encodes args.
        let encoded = func.abi_encode_input(&dyn_values).map_err(|e| EvmError::Encode {
            category: Cow::Borrowed("abi_encode_input"),
            detail_for_log: format!("{e}"),
        })?;
        Ok(Bytes::from(encoded))
    }
    ```

    ```rust
    //!
    //! Pure-function module — no provider, no RPC. Builds the `tx` shape
    //! that Plan 05-02's simulator and Phase 6's signer consume.
    //!
    //! See `crates/executor-evm/tests/normalize.rs` for the per-variant contract.

    use crate::action::{validate_address, validate_calldata, validate_decimal_amount};
    use crate::dyn_abi::encode_call_input;
    use crate::erc20::ERC20_WRITE_ABI;
    use crate::EvmError;
    use alloy::rpc::types::TransactionRequest;
    use alloy_primitives::{Address, Bytes, U256};
    use executor_core::schema::action::{
        Action, ContractCallAction, Erc20ApproveAction, Erc20TransferAction,
        NativeTransferAction, RawCallAction,

    use std::borrow::Cow;
    use std::str::FromStr;

    /// Normalized form of a Phase-4 `Action` ready for simulation/signing.
    /// `Noop` is filtered out earlier — `normalize_action` returns `Ok(None)`.
    #[derive(Debug, Clone)]
    pub struct NormalizedAction {
        pub tx: TransactionRequest,
        pub source: NormalizedActionKind,
        pub selector: Option<[u8; 4]>,
        pub native_value: U256,
        pub erc20_amount: Option<U256>,
    }

    #[derive(Debug, Clone, Copy, PartialEq, Eq)]
    pub enum NormalizedActionKind {
        ContractCall, RawCall, Erc20Transfer, Erc20Approve, NativeTransfer,
    }

    /// Top-level dispatcher. Returns `Ok(None)` for `Noop`.
    pub fn normalize_action(action: &Action) -> Result<Option<NormalizedAction>, EvmError> {
        match action {
            Action::Noop => Ok(None),
            Action::ContractCall(cc) => Ok(Some(normalize_contract_call(cc)?)),
            Action::RawCall(rc) => Ok(Some(normalize_raw_call(rc)?)),
            Action::Erc20Transfer(et) => Ok(Some(normalize_erc20_transfer(et)?)),
            Action::Erc20Approve(ea) => Ok(Some(normalize_erc20_approve(ea)?)),
            Action::NativeTransfer(nt) => Ok(Some(normalize_native_transfer(nt)?)),
        }
    }

    fn parse_address(s: &str, field: &'static str) -> Result<Address, EvmError> {
        // validate_address (Phase 4) is lenient EIP-55 + lowercase. Re-use unchanged.
        validate_address(s).map_err(|_| EvmError::Encode {
            category: Cow::Borrowed("bad_address_to"),
            detail_for_log: format!("{field} = {s} is not a valid address"),
        })
    }

    fn parse_decimal_value(s: &str, field: &'static str) -> Result<U256, EvmError> {
        validate_decimal_amount(s).map_err(|_| EvmError::Encode {
            category: Cow::Borrowed("bad_decimal_value"),
            detail_for_log: format!("{field} = {s} is not a valid decimal U256"),
        })
    }

    pub fn normalize_contract_call(cc: &ContractCallAction) -> Result<NormalizedAction, EvmError> {
        let to = parse_address(&cc.address, "address")?;
        let value = parse_decimal_value(&cc.value, "value")?;
        // D-03: shared encoder (replaces the old "encode-then-discard" Phase-4 path).
        let calldata = encode_call_input(&cc.abi, &cc.function, &cc.args)?;
        let selector = if calldata.len() >= 4 {
            Some([calldata[0], calldata[1], calldata[2], calldata[3]])
        } else {
            None
        };
        let tx = TransactionRequest::default()
            .to(to)
            .input(calldata.clone().into())
            .value(value);
        Ok(NormalizedAction {
            tx, source: NormalizedActionKind::ContractCall,
            selector, native_value: value, erc20_amount: None,
        })
    }

    pub fn normalize_raw_call(rc: &RawCallAction) -> Result<NormalizedAction, EvmError> {
        let to = parse_address(&rc.address, "address")?;
        let value = parse_decimal_value(&rc.value, "value")?;
        // validate_calldata (Phase 4) returns the parsed Bytes.
        let calldata: Bytes = validate_calldata(&rc.data).map_err(|_| EvmError::Encode {
            category: Cow::Borrowed("bad_calldata"),
            detail_for_log: format!("data = {} is not valid 0x-prefixed hex", rc.data),
        })?;
        // P-4: selector is None for sub-4-byte calldata.
        let selector = if calldata.len() >= 4 {
            Some([calldata[0], calldata[1], calldata[2], calldata[3]])
        } else {
            None
        };
        let tx = TransactionRequest::default()
            .to(to)
            .input(calldata.into())
            .value(value);
        Ok(NormalizedAction {
            tx, source: NormalizedActionKind::RawCall,
            selector, native_value: value, erc20_amount: None,
        })
    }

    pub fn normalize_erc20_transfer(et: &Erc20TransferAction) -> Result<NormalizedAction, EvmError> {
        let to_token = parse_address(&et.token, "token")?;
        let recipient = parse_address(&et.to, "to")?;
        let amount = parse_decimal_value(&et.amount, "amount")?;
        // ERC20_WRITE_ABI + encode_call_input pre-pends the 0xa9059cbb selector.
        let calldata = encode_call_input(
            ERC20_WRITE_ABI,
            "transfer",
            &[json!(format!("{recipient:?}")), json!(et.amount.clone())],
        )?;
        debug_assert_eq!(&calldata[..4], &[0xa9, 0x05, 0x9c, 0xbb], "transfer selector");
        let tx = TransactionRequest::default()
            .to(to_token)
            .input(calldata.clone().into())
            .value(U256::ZERO);
        Ok(NormalizedAction {
            tx, source: NormalizedActionKind::Erc20Transfer,
            selector: Some([0xa9, 0x05, 0x9c, 0xbb]),
            native_value: U256::ZERO,
            erc20_amount: Some(amount),
        })
    }

    pub fn normalize_erc20_approve(ea: &Erc20ApproveAction) -> Result<NormalizedAction, EvmError> {
        let to_token = parse_address(&ea.token, "token")?;
        let spender = parse_address(&ea.spender, "spender")?;
        let amount = parse_decimal_value(&ea.amount, "amount")?;
        let calldata = encode_call_input(
            ERC20_WRITE_ABI,
            "approve",
            &[json!(format!("{spender:?}")), json!(ea.amount.clone())],

        debug_assert_eq!(&calldata[..4], &[0x09, 0x5e, 0xa7, 0xb3], "approve selector");
        let tx = TransactionRequest::default()
            .to(to_token)
            .input(calldata.clone().into())
            .value(U256::ZERO);
        Ok(NormalizedAction {
            tx, source: NormalizedActionKind::Erc20Approve,
            selector: Some([0x09, 0x5e, 0xa7, 0xb3]),
            native_value: U256::ZERO,
            erc20_amount: Some(amount),
        })
    }

    pub fn normalize_native_transfer(nt: &NativeTransferAction) -> Result<NormalizedAction, EvmError> {
        let to = parse_address(&nt.to, "to")?;
        let value = parse_decimal_value(&nt.value, "value")?;

            .to(to)
            .input(Bytes::new().into())
            .value(value);

            selector: None,
            native_value: value,
            erc20_amount: None,
        })
    }
    ```
