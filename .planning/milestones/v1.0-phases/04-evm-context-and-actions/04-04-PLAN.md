    use crate::EvmError;
    use alloy_primitives::U256;

    /// D-10: U256 max precision is 78 digits — cap decimals to ≤ 77.
    const MAX_DECIMALS: u8 = 77;

    pub fn parse_units(amount: &str, decimals: u8) -> Result<U256, EvmError> {
        if decimals > MAX_DECIMALS {
            return Err(EvmError::Encode {
                category: "decimals_out_of_range",
                detail_for_log: format!("decimals {decimals} exceeds U256 precision cap {MAX_DECIMALS}"),
            });
        }
        if amount.starts_with('-') {
            return Err(EvmError::Encode {
                category: "amount_negative",
                detail_for_log: "amount must be non-negative".into(),
            });
        }
        let (int_part, frac_part) = match amount.find('.') {
            Some(i) => (&amount[..i], &amount[i+1..]),
            None => (amount, ""),
        };
        if frac_part.len() > decimals as usize {
            return Err(EvmError::Encode {
                category: "amount_overflow_fraction",
                detail_for_log: format!(
                    "fractional digits ({}) exceed decimals ({decimals})",
                    frac_part.len()
                ),
            });
        }
        // Validate digit-only
        let valid_digits = |s: &str| s.chars().all(|c| c.is_ascii_digit());
        if !valid_digits(int_part) && !int_part.is_empty() {
            return Err(EvmError::Encode {
                category: "amount_not_decimal",
                detail_for_log: format!("integer part {int_part:?} must be digits"),
            });
        }
        if !valid_digits(frac_part) && !frac_part.is_empty() {
            return Err(EvmError::Encode {
                category: "amount_not_decimal",
                detail_for_log: format!("fractional part {frac_part:?} must be digits"),
            });
        }
        // Normalize: pad fraction with zeros to `decimals`.
        let int_part = if int_part.is_empty() { "0" } else { int_part };
        let mut combined = String::with_capacity(int_part.len() + decimals as usize);
        combined.push_str(int_part);
        combined.push_str(frac_part);
        for _ in frac_part.len()..(decimals as usize) {
            combined.push('0');
        }
        U256::from_str_radix(&combined, 10).map_err(|e| EvmError::Encode {
            category: "amount_overflow_u256",
            detail_for_log: format!("{e}"),
        })
    }

    pub fn format_units(value: U256, decimals: u8) -> Result<String, EvmError> {
        if decimals > MAX_DECIMALS {
            return Err(EvmError::Encode {
                category: "decimals_out_of_range",
                detail_for_log: format!("decimals {decimals} exceeds {MAX_DECIMALS}"),
            });
        }
        if decimals == 0 {
            return Ok(value.to_string());
        }
        let s = value.to_string();
        let dec = decimals as usize;
        let (int_part, frac_part) = if s.len() <= dec {
            ("0".to_string(), format!("{s:0>width$}", width = dec))
        } else {
            let split = s.len() - dec;
            (s[..split].to_string(), s[split..].to_string())
        };
        let frac_trimmed = frac_part.trim_end_matches('0');
        if frac_trimmed.is_empty() { Ok(int_part) }
        else { Ok(format!("{int_part}.{frac_trimmed}")) }
    }

    use crate::EvmError;
    use alloy_primitives::Address;
    use std::str::FromStr;

    pub const ZERO_ADDRESS: &str = "0x0000000000000000000000000000000000000000";

    pub fn is_address(s: &str) -> bool {
        if s.len() != 42 || !s.starts_with("0x") { return false; }
        // Strict EIP-55 path
        if Address::parse_checksummed(s, None).is_ok() { return true; }
        // All-lower / all-upper fallback
        let body = &s[2..];
        let all_lower = body.chars().all(|c| !c.is_ascii_alphabetic() || c.is_ascii_lowercase());
        let all_upper = body.chars().all(|c| !c.is_ascii_alphabetic() || c.is_ascii_uppercase());
        if (all_lower || all_upper) && Address::from_str(s).is_ok() { return true; }
        false
    }

    pub fn checksum(s: &str) -> Result<String, EvmError> {
        let addr = if let Ok(a) = Address::parse_checksummed(s, None) { a }
        else {
            let body = &s.trim_start_matches("0x");
            let all_lower = body.chars().all(|c| !c.is_ascii_alphabetic() || c.is_ascii_lowercase());
            let all_upper = body.chars().all(|c| !c.is_ascii_alphabetic() || c.is_ascii_uppercase());
            if !(all_lower || all_upper) {
                return Err(EvmError::Encode {
                    category: "bad_address",
                    detail_for_log: "address looks checksummed but checksum is invalid".into(),
                });
            }
            Address::from_str(s).map_err(|e| EvmError::Encode {
                category: "bad_address",
                detail_for_log: format!("{e}"),
            })?
        };
        Ok(addr.to_checksum(None))
    }
