    ```rust
    //! ERC20 read helpers — D-06. All six are thin wrappers around
    //! `read::read_contract` with a bundled OZ-compatible ABI fragment.

    use crate::{EvmError, EvmConfig};
    use crate::read::{read_contract, ReadContractInput, BlockTag};
    use alloy::providers::DynProvider;
    use std::sync::Arc;

    /// OZ-compatible ERC20 ABI fragment. Bundled as a static string so the
    /// agent never has to supply one. Selector-stable across major impls.
    pub const ERC20_ABI: &str = r#"[
        {"type":"function","name":"balanceOf","inputs":[{"name":"account","type":"address"}],"outputs":[{"name":"","type":"uint256"}],"stateMutability":"view"},
        {"type":"function","name":"allowance","inputs":[{"name":"owner","type":"address"},{"name":"spender","type":"address"}],"outputs":[{"name":"","type":"uint256"}],"stateMutability":"view"},
        {"type":"function","name":"decimals","inputs":[],"outputs":[{"name":"","type":"uint8"}],"stateMutability":"view"},
        {"type":"function","name":"symbol","inputs":[],"outputs":[{"name":"","type":"string"}],"stateMutability":"view"},
        {"type":"function","name":"name","inputs":[],"outputs":[{"name":"","type":"string"}],"stateMutability":"view"},
        {"type":"function","name":"totalSupply","inputs":[],"outputs":[{"name":"","type":"uint256"}],"stateMutability":"view"}
    ]"#;

    macro_rules! erc20_helper {
        ($name:ident, $fn:literal, $($arg:ident : &str),* ; $tag:ident) => {
            pub async fn $name(
                provider: Arc<DynProvider>,
                cfg: &EvmConfig,
                token: &str,
                $($arg: &str,)*
                $tag: BlockTag,
            ) -> Result<serde_json::Value, EvmError> {
                read_contract(provider, cfg, ReadContractInput {
                    address: token.into(),
                    abi_json: ERC20_ABI.into(),
                    function: $fn.into(),
                    args: vec![$( serde_json::Value::String($arg.into()) ),*],
                    block_tag: $tag,
                }).await
            }
        };
        // Variant for zero-arg helpers.
        ($name:ident, $fn:literal ; $tag:ident) => {
            pub async fn $name(
                provider: Arc<DynProvider>,
                cfg: &EvmConfig,
                token: &str,
                $tag: BlockTag,
            ) -> Result<serde_json::Value, EvmError> {
                read_contract(provider, cfg, ReadContractInput {
                    address: token.into(),
                    abi_json: ERC20_ABI.into(),
                    function: $fn.into(),
                    args: vec![],
                    block_tag: $tag,
                }).await
            }
        };
    }

    erc20_helper!(erc20_balance_of, "balanceOf", account: &str; block_tag);
    erc20_helper!(erc20_allowance,  "allowance", owner: &str, spender: &str; block_tag);
    erc20_helper!(erc20_decimals,    "decimals"; block_tag);
    erc20_helper!(erc20_symbol,      "symbol"; block_tag);
    erc20_helper!(erc20_name,        "name"; block_tag);
    erc20_helper!(erc20_total_supply,"totalSupply"; block_tag);

    #[cfg(test)]
    mod tests {
        use super::*;
        use alloy_json_abi::JsonAbi;
        #[test]
        fn erc20_abi_parses_and_contains_six_functions() {
            let abi: JsonAbi = serde_json::from_str(ERC20_ABI).unwrap();
            for f in ["balanceOf", "allowance", "decimals", "symbol", "name", "totalSupply"] {
                assert!(abi.function(f).map(|s| !s.is_empty()).unwrap_or(false),
                    "missing fn: {f}");
            }
        }
    }
    ```

    ```rust
    use crate::{EvmConfig, EvmError};
    use alloy::providers::{DynProvider, Provider};
    use alloy_eips::BlockId;     // or alloy::rpc::types::BlockNumberOrTag — match alloy 2.0 path
    use alloy_primitives::Address;
    use std::sync::Arc;
    use std::str::FromStr;

    /// `ctx.evm.readNative.balance(account, blockTag?)` — returns wei as decimal string (D-03).
    pub async fn native_balance(
        provider: Arc<DynProvider>,
        cfg: &EvmConfig,
        account: &str,
        block_tag: crate::read::BlockTag,
    ) -> Result<serde_json::Value, EvmError> {
        let addr = parse_address(account)?;
        let block_id: BlockId = block_tag.into();   // implement From<BlockTag> for BlockId in read.rs
        let bal = tokio::time::timeout(
            cfg.call_timeout,
            provider.get_balance(addr).block_id(block_id),
        ).await
            .map_err(|_| EvmError::Timeout)?
            .map_err(|e| EvmError::Transport { detail_for_log: e.to_string() })?;
        Ok(serde_json::Value::String(bal.to_string()))   // U256: Display = base-10
    }

    /// `ctx.evm.readNative.blockNumber()` — returns the current block number as JS Number.
    pub async fn native_block_number(
        provider: Arc<DynProvider>,
        cfg: &EvmConfig,
    ) -> Result<serde_json::Value, EvmError> {
        let n = tokio::time::timeout(cfg.call_timeout, provider.get_block_number())
            .await
            .map_err(|_| EvmError::Timeout)?
            .map_err(|e| EvmError::Transport { detail_for_log: e.to_string() })?;
        Ok(serde_json::Value::Number(n.into()))
    }

    fn parse_address(s: &str) -> Result<Address, EvmError> {
        // Use the same lenient EIP-55-or-lowercase parse from read.rs (Plan 04-01)
        // — refactor a shared `crate::dyn_abi::parse_address` if not already public.
        Address::parse_checksummed(s, None).or_else(|_| Address::from_str(s))
            .map_err(|e| EvmError::Encode {
                category: "bad_address",
                detail_for_log: format!("{e}"),
            })
    }

    #[cfg(test)]
    mod tests {
        use super::*;
        // Shape tests — no anvil here, just type-shape pinning.
        #[test]
        fn native_balance_emits_string_for_uint256() {
            // Smoke: a U256 stringifies to base-10 (no `0x` prefix).
            let v = alloy_primitives::U256::from(1_000_000u64).to_string();
            assert_eq!(v, "1000000");
        }
    }
    ```
