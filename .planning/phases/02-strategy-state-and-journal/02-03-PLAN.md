    ```rust
    #[tokio::test]
    async fn run_status_schema_includes_future_variants() -> Result<()> {
        // D-08a: prove the RunStatus JSON Schema golden carries all 7 snake_case
        // variants, including the 3 future-reserved ones that Phase 2 MUST NOT
        // emit (Canceled, SimulationDenied, PolicyDenied) — Phase 5/6 can rely
        // on these names not drifting.
        let path = std::path::Path::new("../executor-core/tests/schemas/RunStatus.json");
        let text = std::fs::read_to_string(path)
            .unwrap_or_else(|e| panic!("read {path:?}: {e}"));
        let v: Value = serde_json::from_str(&text)?;

        // schemars 1.2 emits an `enum` array somewhere under the root — the exact
        // shape is either top-level `{ "type": "string", "enum": [...] }` or
        // nested under `$defs`. Walk the value and collect any `enum` arrays.
        fn collect_enums(v: &Value, out: &mut Vec<Vec<String>>) {
            match v {
                Value::Object(m) => {
                    if let Some(e) = m.get("enum").and_then(|x| x.as_array()) {
                        out.push(e.iter().filter_map(|s| s.as_str().map(String::from)).collect());
                    }
                    for (_k, val) in m { collect_enums(val, out); }
                }
                Value::Array(a) => for x in a { collect_enums(x, out); },
                _ => {}
            }
        }
        let mut all_enums = Vec::new();
        collect_enums(&v, &mut all_enums);
    ```
