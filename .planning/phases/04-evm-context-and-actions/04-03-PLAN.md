    ```rust
    use crate::EvmError;
    use alloy_primitives::{Address, Bytes, U256};
    use alloy_json_abi::JsonAbi;
    use std::str::FromStr;
    use executor_core::schema::action::MAX_ABI_BYTES;

    pub fn validate_address(s: &str) -> Result<Address, EvmError> {
        // Lenient: try EIP-55 first; fall back to from_str only for all-lower or all-upper.
        if let Ok(a) = Address::parse_checksummed(s, None) { return Ok(a); }
        let trimmed = s.trim_start_matches("0x");
        let is_all_lower = trimmed.chars().all(|c| !c.is_ascii_alphabetic() || c.is_ascii_lowercase());
        let is_all_upper = trimmed.chars().all(|c| !c.is_ascii_alphabetic() || c.is_ascii_uppercase());
        if is_all_lower || is_all_upper {
            return Address::from_str(s).map_err(|e| EvmError::Encode {
                category: "bad_address",
                detail_for_log: format!("{e}"),
            });
        }
        Err(EvmError::Encode {
            category: "bad_address",
            detail_for_log: format!("address looks checksummed but checksum is invalid: {s}"),
        })
    }

    pub fn validate_calldata(s: &str) -> Result<Bytes, EvmError> {
        if !s.starts_with("0x") {
            return Err(EvmError::Encode {
                category: "bad_calldata",
                detail_for_log: format!("calldata must be 0x-prefixed: {s}"),
            });
        }
        let body = &s[2..];
        if body.len() % 2 != 0 {
            return Err(EvmError::Encode {
                category: "bad_calldata",
                detail_for_log: "calldata hex length must be even".into(),
            });
        }
        Bytes::from_str(s).map_err(|e| EvmError::Encode {
            category: "bad_calldata",
            detail_for_log: format!("{e}"),
        })
    }

    pub fn validate_decimal_amount(s: &str) -> Result<U256, EvmError> {
        if s.starts_with('-') {
            return Err(EvmError::Encode {
                category: "bad_decimal",
                detail_for_log: "amount must be non-negative".into(),
            });
        }
        if s.starts_with("0x") || s.starts_with("0X") {
            return Err(EvmError::Encode {
                category: "bad_decimal",
                detail_for_log: "amount must be decimal (no 0x prefix); use ctx.units.parseUnits".into(),
            });
        }
        U256::from_str_radix(s, 10).map_err(|e| EvmError::Encode {
            category: "bad_decimal",
            detail_for_log: format!("{e}"),
        })
    }

    pub fn validate_abi_size(s: &str) -> Result<(), EvmError> {
        if s.len() > MAX_ABI_BYTES {
            return Err(EvmError::Encode {
                category: "abi_oversize",
                detail_for_log: format!("abi size {} exceeds {}", s.len(), MAX_ABI_BYTES),
            });
        }
        Ok(())
    }

    /// Dry-run ABI encode at builder time (D-09). Returns Ok(()) when the
    /// args match the function signature. Bytes are DISCARDED — Phase 5 owns
    /// canonical encoding for transaction calldata.
    pub fn dry_run_abi_encode(
        abi: &str,
        function: &str,
        args: &[serde_json::Value],
    ) -> Result<(), EvmError> {
        validate_abi_size(abi)?;
        let parsed: JsonAbi = serde_json::from_str(abi).map_err(|e| EvmError::Decode {
            category: "abi_parse",
            detail_for_log: format!("{e}"),
        })?;
        let fns = parsed.function(function);
        let candidates: &[alloy_json_abi::Function] = match fns {
            Some(fs) if !fs.is_empty() => fs,
            _ => return Err(EvmError::Decode {
                category: "abi_function_missing",
                detail_for_log: format!("abi does not contain function {function}"),
            }),
        };
        // Pick by arg count
        let func = candidates.iter().find(|f| f.inputs.len() == args.len()).ok_or(
            EvmError::Encode {
                category: "abi_arg_count",
                detail_for_log: format!("no overload of {function} accepts {} args", args.len()),
            }
        )?;
        // Encode and discard
        let dyn_values: Vec<_> = func.inputs.iter().zip(args).map(|(p, a)| {
            let ty: alloy_dyn_abi::DynSolType = p.selector_type_name().parse().map_err(|e| {
                EvmError::Encode { category: "abi_type_parse", detail_for_log: format!("{e}") }
            })?;
            crate::dyn_abi::js_value_to_dyn_sol(a, &ty)
        }).collect::<Result<_, _>>()?;
        let _bytes = func.abi_encode_input(&dyn_values).map_err(|e| EvmError::Encode {
            category: "abi_encode_input",
            detail_for_log: format!("{e}"),
        })?;
        Ok(())
    }

    mod tests { /* test cases per <behavior> Tests 7-12 */ }
    ```
