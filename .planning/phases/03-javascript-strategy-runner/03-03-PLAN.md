The validation rule for the strategy's return value:
1. If JSON is `Value::String("noop")` → `StrategyOutcome::Noop`.
2. If JSON is `Value::Array(items)` → for each item, deserialize as `Action`. On any failure, `RuntimeError::InvalidOutput { detail: format!("invalid action at index {i}: {msg}") }`. On success, `StrategyOutcome::Actions { actions }`. (Empty array `[]` is allowed and produces `Actions { actions: vec![] }`.)
3. Otherwise (Number, Bool, Object, Null) → `RuntimeError::InvalidOutput { detail: format!("expected 'noop' or Action[], got {type_name}") }`.

This validator lives in `executor-mcp::tools::validate_strategy_output` (or as a free function in a new `executor-core::schema::action` helper — recommended: keep it in `executor-mcp::tools` so it can produce executor-mcp-facing error envelopes directly, but since the deserialisation is via `serde_json::from_value::<Action>` the logic is two lines and could go either place; pick whichever produces cleaner test ergonomics).

The strategy_run handler implements the 8-step flow from 03-RESEARCH.md `Source-Read & Journal Capture Strategy → Run lifecycle`:

```rust
async fn strategy_run(
    &self,
    Parameters(input): Parameters<StrategyRunInput>,
) -> Result<CallToolResult, McpError> {
    // STEP 1: validate input (D-09 — reuse Phase-2 helper)
    validate_strategy_id_format(&input.strategy_id).map_err(invalid_params)?;

    // STEP 2: load strategy + check deleted
    let state = self.state.clone();
    let strategy_id_for_load = input.strategy_id.clone();
    let strategy = tokio::task::spawn_blocking(move || {
        let store = state.blocking_lock();
        store.get_strategy_by_id(&strategy_id_for_load)
    })
    .await
    .map_err(|e| storage_error(format!("spawn_blocking join: {e}")))?
    .map_err(map_state_error)?
    .ok_or_else(|| map_state_error(StateError::NotFound(format!("strategy {}", input.strategy_id))))?;

    if strategy.deleted_at.is_some() {
        return Err(strategy_deleted(&strategy.id));
    }

    // STEP 3: insert run (Queued)
    let state = self.state.clone();
    let strategy_id_for_run = strategy.id.clone();
    let run_id = tokio::task::spawn_blocking(move || {
        let mut store = state.blocking_lock();
        store.insert_run(&strategy_id_for_run, RunStatus::Queued)
    })
    .await
    .map_err(|e| storage_error(format!("spawn_blocking join: {e}")))?
    .map_err(map_state_error)?;

    // STEP 4: transition Queued → Running (D-12)
    let state = self.state.clone();
    let run_id_for_running = run_id.clone();
    tokio::task::spawn_blocking(move || {
        let mut store = state.blocking_lock();
        store.update_run_status_with_transition(&run_id_for_running, RunStatus::Queued, RunStatus::Running)
    })
    .await
    .map_err(|e| storage_error(format!("spawn_blocking join: {e}")))?
    .map_err(map_state_error)?;

    // Capture started_at by reading the run row back. (Alternative: have insert_run return the started_at; defer
    // to executor — pick whichever is simpler. RESEARCH suggests simply reading the row at the end.)

    // STEP 5: spawn_blocking { Sandbox::execute + RuntimeContext::flush }
    //   This block does NOT acquire the state mutex during JS execution; flush
    //   acquires it once after execute returns (Pitfall 2 / Pitfall 4 carry-over).
    let state_for_run = self.state.clone();
    let strategy_source = strategy.source.clone();
    let strategy_id_for_ctx = strategy.id.clone();
    let strategy_name_for_ctx = strategy.name.clone();
    let run_id_for_ctx = run_id.clone();
    let exec_result: Result<serde_json::Value, strategy_js::RuntimeError> =
        tokio::task::spawn_blocking(move || -> Result<serde_json::Value, strategy_js::RuntimeError> {
            let mut runtime_ctx = strategy_js::RuntimeContext::new(
                state_for_run.clone(),
                strategy_id_for_ctx,
                strategy_name_for_ctx,
                run_id_for_ctx,
                strategy_js::RuntimeContext::default_clock(),
            );
            let r = strategy_js::Sandbox::execute(&strategy_source, &mut runtime_ctx);
            // Flush even on Err — we want logs + source-read marker recorded
            // regardless of success/failure (STJ-03 / STJ-04).
            // Map flush errors into RuntimeError::Exception so the outer match
            // path can map both cleanly. Document: a flush failure here is
            // already a storage failure, but presenting it as RuntimeError
            // gives the agent a single error envelope path.
            //
            // Alternative: surface flush errors as Storage(_) directly via a
            // sibling result Tuple. Pick the cleaner pattern; document in
            // SUMMARY.
            if let Err(flush_err) = runtime_ctx.flush() {
                tracing::warn!(?flush_err, "RuntimeContext::flush failed after execute");
            }
            r
        })
        .await
        .map_err(|e| storage_error(format!("spawn_blocking join: {e}")))?;

    // STEP 6: validate output + record action_outcome
    //   - On Ok(json): try validate_strategy_output → Outcome { Noop | Actions(Vec<Action>) }
    //   - On Err(RuntimeError): map to STRATEGY_RUNTIME_ERROR / STRATEGY_INVALID_OUTPUT MCP error
    let outcome_for_response: Result<StrategyOutcome, McpError> = match exec_result {
        Ok(json) => match validate_strategy_output(&json) {
            Ok(out) => {
                // Record success outcome row.
                record_outcome_row(&self.state, &run_id, &out, &json).await?;
                Ok(out)
            }
            Err(detail) => {
                // Record validation_error outcome row.
                record_invalid_output(&self.state, &run_id, &detail, &json).await?;
                // Transition to Failed.
                transition_to_failed(&self.state, &run_id).await?;
                return Err(strategy_invalid_output(detail, &run_id));
            }
        },
        Err(strategy_js::RuntimeError::InvalidOutput { detail }) => {
            record_invalid_output(&self.state, &run_id, &detail, &serde_json::Value::Null).await?;
            transition_to_failed(&self.state, &run_id).await?;
            return Err(strategy_invalid_output(detail, &run_id));
        }
        Err(other_runtime_err) => {
            // Timeout / Oom / StackOverflow / Exception / EngineInit
            record_runtime_error(&self.state, &run_id, &other_runtime_err).await?;
            transition_to_failed(&self.state, &run_id).await?;
            return Err(map_runtime_error(other_runtime_err, &run_id));
        }
    };

    // STEP 7: transition Running → Succeeded
    let state = self.state.clone();
    let run_id_for_done = run_id.clone();
    tokio::task::spawn_blocking(move || {
        let mut store = state.blocking_lock();
        store.update_run_status_with_transition(&run_id_for_done, RunStatus::Running, RunStatus::Succeeded)
    })
    .await
    .map_err(|e| storage_error(format!("spawn_blocking join: {e}")))?
    .map_err(map_state_error)?;

    // STEP 8: re-read the run row to populate started_at + finished_at, then
    //   shape and return the response.
    let outcome = outcome_for_response?;   // Ok at this point
    let state = self.state.clone();
    let run_id_for_get = run_id.clone();
    let run = tokio::task::spawn_blocking(move || {
        let store = state.blocking_lock();
        store.get_run(&run_id_for_get)
    })
    .await
    .map_err(|e| storage_error(format!("spawn_blocking join: {e}")))?
    .map_err(map_state_error)?
    .ok_or_else(|| storage_error("run row vanished between insert and get"))?;

    json_result(&StrategyRunResponse {
        run_id: run.id,
        strategy_id: run.strategy_id,
        status: run.status,
        started_at: run.started_at,
        finished_at: run.finished_at.unwrap_or_default(),
        outcome,
    })
}
```

    ```rust
        let sid_for_load = input.strategy_id.clone();
        let strategy: Strategy = tokio::task::spawn_blocking(move || {
            let store = state.blocking_lock();
            store.get_strategy_by_id(&sid_for_load)

            let mut store = state.blocking_lock();
            store.insert_run(&sid_for_run, RunStatus::Queued)

                let mut runtime_ctx = RuntimeContext::new(
                    state_for_run,
                    sid_for_ctx,
                    sname_for_ctx,
                    rid_for_ctx,
                    RuntimeContext::default_clock(),
                );
                let r = Sandbox::execute(&source, &mut runtime_ctx);
                if let Err(flush_err) = runtime_ctx.flush() {
                    tracing::warn!(?flush_err, "RuntimeContext::flush failed");
                }
                r
    ```

    ```rust
    async fn read_journal(
        uri: String,
        run_id: String,
        state: Arc<tokio::sync::Mutex<StateStore>>,
    ) -> Result<ReadResourceResult, McpError> {
        // Boundary check: ULID is 26 chars, Crockford alphabet. We use a permissive
        // shape check (length + ASCII alphanumeric) — full Crockford validation
        // would require excluding I/L/O/U which is overkill for resource read.
        if run_id.len() != 26 || !run_id.chars().all(|c| c.is_ascii_alphanumeric()) {
            return Err(McpError::resource_not_found(
                format!("malformed run id in uri: {uri}"),
                Some(json!({ "uri": uri, "code": "malformed_id" })),
            ));
        }

        let rid_owned = run_id.clone();
        let (sources, actions, logs) = tokio::task::spawn_blocking(move || {
            let store = state.blocking_lock();
            // Verify run exists first.
            let exists = store.get_run(&rid_owned)?;
            if exists.is_none() {
                return Ok::<_, executor_state::StateError>((None, None, None));
            }
            let s = store.list_source_reads_for_run(&rid_owned)?;
            let a = store.list_actions_for_run(&rid_owned)?;
            let l = store.list_logs_for_run(&rid_owned)?;
            Ok((Some(s), Some(a), Some(l)))
        })
        .await
        .map_err(|e| storage_error(format!("spawn_blocking join: {e}")))?
        .map_err(map_state_error)?;

        let (sources, actions, logs) = match (sources, actions, logs) {
            (Some(s), Some(a), Some(l)) => (s, a, l),
            _ => {
                return Err(McpError::resource_not_found(
                    format!("run {uri} not found"),
                    Some(json!({ "uri": uri })),
                ));
            }
        };

        // Convert journal rows to JSON. Each entry struct already derives Serialize-able
        // shape (we map to the wire form here to keep the public API stable).
        let body = serde_json::json!({
            "run_id": run_id,
            "source_reads": sources.iter().map(|s| serde_json::json!({
                "id": s.id, "kind": s.kind, "target": s.target,
                "payload_json": s.payload_json,
                "recorded_at": s.recorded_at,
            })).collect::<Vec<_>>(),
            "actions": actions.iter().map(|a| serde_json::json!({
                "id": a.id,
                // Use serde to round-trip the enum to its wire form. The
                // `JournalActionOutcome` derive uses #[serde(rename_all = "snake_case")]
                // so `Succeeded`/`SimulationFailure`/etc. become exact wire strings.
                // Do NOT use `format!("{:?}", a.outcome).to_lowercase()` — that form
                // produces "simulationfailure" instead of "simulation_failure" and would
                // silently corrupt the journal payload + break the schema golden.
                "outcome": serde_json::to_value(&a.outcome)
                    .map_err(|e| storage_error(format!("serialize outcome: {e}")))?,
                "payload_json": a.payload_json,
                "recorded_at": a.recorded_at,
            })).collect::<Vec<_>>(),
            "logs": logs.iter().map(|l| serde_json::json!({
                "id": l.id, "message": l.message,
                "recorded_at": l.recorded_at,
            })).collect::<Vec<_>>(),
        });
        let body_text = serde_json::to_string(&body)
            .map_err(|e| storage_error(format!("serialize journal: {e}")))?;
        Ok(ReadResourceResult::new(vec![
            ResourceContents::text(body_text, uri).with_mime_type("application/json"),
        ]))
    }
    ```
